class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int N = nums.size();
        int recur = 1;
        for (int i=0; i<N-1; i++){
            if (i>1 && nums[i]!=nums[i-1]){
                recur = 1;
            }
            if (nums[i]==nums[i+1]){
                recur+=1;
                if(recur==3){
                    for(int j=i+1; j<nums.size()-1; j++){
                        nums[j] = nums[j+1];
                    }
                    recur -= 1 ;
                    nums.pop_back();
                }
            }
            else {
                recur = 1;
            }
        }
        return nums.size();
    }
};

class Solution {
public:
    int work(vector<int>& nums, int k) {
        int len = 0;
        for(auto num : nums)
            if(len < k || nums[len-k] != num)
                nums[len++] = num;
        return len;
    }
    int removeDuplicates(vector<int>& nums) {
        return work(nums, 2);
    }
};
